for i in range(1,10):
    line = ''
    for j in range(1,10):
        if i>j :
            line = '{0} {1} {2:<4}'.format(' ',' ',  ' ')+line
        else:
            line = line+'{0}*{1}={2:<4}'.format(i,j,j*i)
            #print ('=='+line)
    print (line)
#            line = '{0} {1} {2:<4}'.format(' ',' ',  ' ')+line
#当i<j时，打印空格，当i<j时，拉上字符。
